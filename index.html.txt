  // Advanced Global State Management
  let state = {
    patients: JSON.parse(localStorage.getItem('dih_patients') || '[]'),
    visits: JSON.parse(localStorage.getItem('dih_visits') || '[]'),
    pharmacy: JSON.parse(localStorage.getItem('dih_pharmacy') || '[{"name":"Paracetamol 500mg","unit":"Box","qty":150,"price":2},{"name":"Amoxicillin 500mg","unit":"Box","qty":80,"price":4.5},{"name":"Ceftriaxone 1g Inj","unit":"Vial","qty":45,"price":3}]'),
    billing: JSON.parse(localStorage.getItem('dih_billing') || '[]'),
    labPending: JSON.parse(localStorage.getItem('dih_lab_pending') || '[]')
  };

  function syncStorage() {
    localStorage.setItem('dih_patients', JSON.stringify(state.patients));
    localStorage.setItem('dih_visits', JSON.stringify(state.visits));
    localStorage.setItem('dih_pharmacy', JSON.stringify(state.pharmacy));
    localStorage.setItem('dih_billing', JSON.stringify(state.billing));
    localStorage.setItem('dih_lab_pending', JSON.stringify(state.labPending));
    updateUI();
  }

  // Navigation
  function switchTab(tabId) {
    document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
    document.getElementById('tab-' + tabId).classList.add('active');
    
    document.querySelectorAll('.sidebar-link').forEach(l => l.classList.remove('active'));
    const link = document.getElementById('link-' + tabId);
    if(link) link.classList.add('active');
    
    if(window.innerWidth < 1024) toggleSidebar(false);
    
    // Lazy Load Data
    if(tabId === 'opd') populatePatientLists();
    if(tabId === 'pharmacy') renderPharmacy();
    if(tabId === 'lab') renderLab();
    if(tabId === 'billing') renderBilling();
    if(tabId === 'dashboard') renderDashboardTable();
  }

  function toggleSidebar(force) {
    const sb = document.getElementById('sidebar');
    if(force === false) sb.classList.remove('open');
    else sb.classList.toggle('open');
  }

  // UI Updates
  function updateUI() {
    document.getElementById('stat-patients').innerText = state.patients.length;
    let rev = state.billing.filter(b => b.status === 'Paid').reduce((acc, b) => acc + parseFloat(b.total || 0), 0);
    document.getElementById('stat-revenue').innerText = '$' + rev.toFixed(2);
    document.getElementById('stat-emergency').innerText = state.patients.filter(p => p.department === 'Emergency' && p.status === 'Active').length;
    document.getElementById('stat-lab').innerText = state.labPending.length;
    document.getElementById('current-time').innerText = new Date().toLocaleString();
  }

  function renderDashboardTable() {
      const tbody = document.getElementById('dashboard-patient-table');
      const recent = state.patients.slice(-5).reverse();
      if(recent.length === 0) {
          tbody.innerHTML = '<tr><td colspan="4" class="py-10 text-center text-slate-400">No recent activity</td></tr>';
          return;
      }
      tbody.innerHTML = recent.map(p => `
        <tr class="border-b last:border-0">
            <td class="py-4">
                <p class="font-bold text-slate-800">${p.name}</p>
                <p class="text-[10px] text-slate-400">${p.id}</p>
            </td>
            <td class="py-4"><span class="text-xs px-2 py-1 bg-slate-100 rounded">${p.department}</span></td>
            <td class="py-4"><span class="w-2 h-2 rounded-full inline-block bg-green-500 mr-2"></span> Active</td>
            <td class="py-4"><button onclick="switchTab('opd')" class="text-blue-600 hover:underline">Manage</button></td>
        </tr>
      `).join('');
  }

  // Reception Functions
  function savePatient() {
    const name = document.getElementById('reg-name').value;
    const age = document.getElementById('reg-age').value;
    const dept = document.getElementById('reg-dept').value;
    const fee = document.getElementById('reg-fee').value;
    
    if(!name || !age) {
        showNotify("Fadlan buuxi Magaca iyo Da'da", "error");
        return;
    }

    const pId = 'DIH-' + Math.floor(1000 + Math.random() * 9000);
    const patient = {
      id: pId,
      name, age, 
      sex: document.getElementById('reg-sex').value,
      phone: document.getElementById('reg-phone').value,
      department: dept,
      complaint: document.getElementById('reg-complaint').value,
      status: 'Active',
      regFee: parseFloat(fee || 0),
      timestamp: new Date().toISOString()
    };

    state.patients.push(patient);
    
    // Create initial bill
    state.billing.push({
        id: 'INV-' + Date.now().toString().slice(-6),
        patientId: pId,
        patientName: name,
        regFee: parseFloat(fee || 0),
        labFee: 0,
        pharmFee: 0,
        total: parseFloat(fee || 0),
        status: 'Unpaid',
        date: new Date().toISOString().split('T')[0]
    });

    syncStorage();
    showNotify("Bukaanka waa la diiwaangeliyey!", "success");
    
    document.querySelectorAll('#tab-reception input, #tab-reception textarea').forEach(i => i.value = '');
    document.getElementById('reg-fee').value = 5;
    switchTab('dashboard');
  }
